<script>
  function showRegister() {
    document.getElementById('slider').classList.add('show-register');
  }
  function showLogin() {
    document.getElementById('slider').classList.remove('show-register');
  }

  /* ── Login ── */
  function handleLogin() {
    const user = document.getElementById('login-user').value.trim();
    const pass = document.getElementById('login-pass').value;
    const err  = document.getElementById('login-error');

    if (!user || !pass) {
      err.textContent = 'Please fill in both fields.';
      return;
    }
    err.textContent = '';
    alert(`Signing in as "${user}" …`);
  }

  /* ── Password rules ── */
  function checkRules(pass, username) {
    return {
      len:     pass.length >= 12,
      upper:   /[A-Z]/.test(pass),
      num:     /[0-9]/.test(pass),
      special: /[^A-Za-z0-9]/.test(pass),
      nouser:  username.length === 0 || !pass.toLowerCase().includes(username.toLowerCase())
    };
  }

  function validateRegForm() {
    const username = document.getElementById('reg-user').value.trim();
    const pass     = document.getElementById('reg-pass').value;
    const email    = document.getElementById('reg-email').value.trim();
    const r        = checkRules(pass, username);

    setRule('r-len',     r.len);
    setRule('r-upper',   r.upper);
    setRule('r-num',     r.num);
    setRule('r-special', r.special);
    setRule('r-nouser',  r.nouser);

    const passInput = document.getElementById('reg-pass');
    if (pass.length > 0) {
      const allOk = r.len && r.upper && r.num && r.special && r.nouser;
      passInput.classList.toggle('valid',   allOk);
      passInput.classList.toggle('invalid', !allOk);
    } else {
      passInput.classList.remove('valid', 'invalid');
    }

    const emailOk = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    const emailEl = document.getElementById('reg-email');
    if (email.length > 0) {
      emailEl.classList.toggle('valid',   emailOk);
      emailEl.classList.toggle('invalid', !emailOk);
      document.getElementById('email-error').textContent = emailOk ? '' : 'Enter a valid email address.';
    } else {
      emailEl.classList.remove('valid', 'invalid');
      document.getElementById('email-error').textContent = '';
    }
  }

  function setRule(id, ok) {
    document.getElementById(id).classList.toggle('ok', ok);
  }

  /* ── Register submit ── */
  function handleRegister() {
    const username = document.getElementById('reg-user').value.trim();
    const email    = document.getElementById('reg-email').value.trim();
    const pass     = document.getElementById('reg-pass').value;
    const err      = document.getElementById('reg-error');

    if (!username) { err.textContent = 'Please choose a username.'; return; }

    const emailOk = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    if (!emailOk)  { err.textContent = 'Please enter a valid email.'; return; }

    const r = checkRules(pass, username);
    if (!r.len)     { err.textContent = 'Password must be at least 12 characters.'; return; }
    if (!r.upper)   { err.textContent = 'Password must contain an uppercase letter.'; return; }
    if (!r.num)     { err.textContent = 'Password must contain a number.'; return; }
    if (!r.special) { err.textContent = 'Password must contain a special character.'; return; }
    if (!r.nouser)  { err.textContent = 'Password must not contain your username.'; return; }

    err.textContent = '';
    alert(`Account created for "${username}" — welcome!`);
  }
</script>
